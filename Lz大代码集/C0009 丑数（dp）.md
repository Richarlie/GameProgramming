# 丑数
# 19116 丑数

# Description

“丑数”是指除了质因子2,3，5，不含其它质因子的正整数，例如由小到大前10个“丑数”为
1, 2, 3, 4, 5, 6, 8, 9, 10, 12, ...
现要求编写一个程序，输出指定第几位的“丑数”。


### 输入格式
第一行为正整数T（T<=10000）, 表示case的数目。
此后T行，每行一个正整数 n (确定该第n个凑数不超过100000000).
### 输出格式

每一个n，输出第n个“丑数”

### 输入样例

3
1
2
9

  
  

### 输出样例

1
2
10

```C++
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

// 动态规划生成丑数
long long nthUglyNumber(int n) {
    vector<long long> dp(n + 1, 0); // dp[i]表示第i个丑数
    dp[1] = 1; // 第一个丑数是1
    int p2 = 1, p3 = 1, p5 = 1; // 分别指向当前可以乘以2、3、5的丑数位置

    for (int i = 2; i <= n; ++i) {
        dp[i] = min({dp[p2] * 2, dp[p3] * 3, dp[p5] * 5}); // 选择最小的丑数
        if (dp[i] == dp[p2] * 2) p2++; // 如果是乘以2得到的，p2指针后移
        if (dp[i] == dp[p3] * 3) p3++; // 如果是乘以3得到的，p3指针后移
        if (dp[i] == dp[p5] * 5) p5++; // 如果是乘以5得到的，p5指针后移
    }

    return dp[n]; // 返回第n个丑数
}

int main() {
	vector<long long> dp(1700, 0); // dp[i]表示第i个丑数
    dp[1] = 1; // 第一个丑数是1
    int p2 = 1, p3 = 1, p5 = 1; // 分别指向当前可以乘以2、3、5的丑数位置
    for (int i = 2; i <= 1700; ++i) {
        dp[i] = min({dp[p2] * 2, dp[p3] * 3, dp[p5] * 5}); // 选择最小的丑数
        if (dp[i] == dp[p2] * 2) p2++; // 如果是乘以2得到的，p2指针后移
        if (dp[i] == dp[p3] * 3) p3++; // 如果是乘以3得到的，p3指针后移
        if (dp[i] == dp[p5] * 5) p5++; // 如果是乘以5得到的，p5指针后移
    }
    int T; // 测试用例数量
    cin >> T;
	for(int i=0;i<T;i++){
		int k;
		cin>>k;
		cout<<dp[k]<<endl;
	}
    return 0;
}
```
