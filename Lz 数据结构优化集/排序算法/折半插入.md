# 折半插入排序
```C++
void Sort(vector<int>&arr){
		for(int i=1;i<arr.size();i++){
			int temp=arr[i];
			int left=0,right=i-1;
			while(left<=right){
				int mid=left+(right-left)/2;
				if(temp<arr[mid]){
					right=mid-1;
				}
				else{
					left=mid+1;
				}
			}
		
			for(int j=i-1;j>=left;j--){
				arr[j+1]=arr[j];
			}
			arr[left]=temp;
			
			for(int j=0;j<arr.size();j++){
				cout<<arr[j]<<" ";
			}
			cout<<endl;
		}
}
```
