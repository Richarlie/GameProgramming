# 折半插入
```C++
void binaryInsertionSort(int arr[], int n) {
    for (int i = 1; i < n; i++) {
        int temp = arr[i];
        int left = 0;
        int right = i - 1;

        // 使用折半查找确定插入位置
        while (left <= right) {
            int mid = left + (right - left) / 2;
            if (temp < arr[mid]) {
                right = mid - 1;
            } else {
                left = mid + 1;
            }
        }

        // 将从插入位置到当前位置的元素向后移动
        for (int j = i - 1; j >= left; j--) {
            arr[j + 1] = arr[j];
        }

        // 插入当前元素
        arr[left] = temp;

        // 输出每趟排序的结果
        for (int k = 0; k < n; k++) {
            cout << arr[k];
            if (k < n - 1) {
                cout << " ";
            }
        }
        cout << endl;
    }
}
```
