# 冒泡排序
```C++
void bubbleSort(int arr[], int n) {
    bool swapped; // 标志位，用于判断是否发生交换
    for (int i = 0; i < n - 1; i++) { // 外层循环控制排序的趟数
        swapped = false; // 每趟开始时，假设没有交换
        for (int j = 0; j < n - 1 - i; j++) { // 内层循环控制每趟的比较次数
            if (arr[j] > arr[j + 1]) { // 如果相邻两个元素逆序
                swap(arr[j], arr[j + 1]); // 交换它们的位置
                swapped = true; // 设置标志位为true，表示发生了交换
            }
        }
        // 输出每趟排序的结果
        for (int k = 0; k < n; k++) {
            cout << arr[k];
            if (k < n - 1) {
                cout << " ";
            }
        }
        cout << endl;
        if (!swapped) { // 如果某趟没有发生交换，说明数组已经有序
            break; // 提前结束排序
        }
    }
}
```
