# 直接插入排序
```C++
void Sort(vector<int>&arr){
		for(int i=1;i<arr.size();i++){
			int temp=arr[i];
			int j=i-1;
			while(j>=0&&arr[j]>temp){
				arr[j+1]=arr[j];
				j-=1;
			}
			arr[j+1]=temp;
		
		for(int i=0;i<arr.size();i++){
			cout<<arr[i]<<" ";
		}
		cout<<endl;
	}
}
```
