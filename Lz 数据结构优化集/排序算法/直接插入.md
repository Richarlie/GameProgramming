# 直接插入排序
```C++
void Sort(int arr[], int n) {
    for (int i = 1; i < n; i++) {
        int temp = arr[i];
        int j = i - 1;
        while (j >= 0 && arr[j] > temp) {
            arr[j + 1] = arr[j];
            j--;
        }
        arr[j + 1] = temp;

        // 输出每趟排序的结果
        for (int k = 0; k < n; k++) {
            cout << arr[k];
            if (k < n - 1) {
                cout << " ";
            }
        }
        cout << endl;
    }
}
```
