# 广度优先
```C
#include <iostream>
#include <cstring>
#include <cstdlib>
#include <queue> // 引入C++的queue容器
using namespace std;

void BFS(ALGraph G, int v) {
    queue<int> Q; // 使用C++的queue容器
    visited[v] = 1;
    VisitFunc(G.vertices[v].data);
    Q.push(v);
    while (!Q.empty()) {
        int u = Q.front();
        Q.pop();
        ArcNode *p = G.vertices[u].firstarc;
        while (p != NULL) {
            int w = p->adjvex;
            if (!visited[w]) {
                visited[w] = 1;
                VisitFunc(G.vertices[w].data);
                Q.push(w);
            }
            p = p->nextarc;
        }
    }
}

void BFSTraverse(ALGraph G) {
    VisitFunc = print;
    for (int i = 0; i < G.vexnum; i++)
        visited[i] = 0;
    for (int i = 0; i < G.vexnum; i++)
        if (!visited[i])
            BFS(G, i);
    printf("\n");
}
```
