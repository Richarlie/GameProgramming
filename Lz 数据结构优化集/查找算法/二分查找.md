# 二分查找
```C
//非递归形式
int Search_Bin(SSTable &ST, ElemType key)
{
    int low = 0, high = ST.length - 1; // 初始化查找范围
    while (low <= high)
    {
        int mid = low + (high - low) / 2; // 避免溢出
        if (ST.elem[mid] == key) // 找到目标元素
        {
            return mid; // 返回位置（从0开始）
        }
        else if (ST.elem[mid] < key) // 目标在右半部分
        {
            low = mid + 1;
        }
        else // 目标在左半部分
        {
            high = mid - 1;
        }
    }
    return -1; // 未找到目标元素
}
```
```c
//递归形式
int Search_Bin_Recursive(SSTable &ST, ElemType key, int low, int high)
{
    if (low > high) // 基本情况：查找范围为空
    {
        return -1; // 未找到目标元素
    }
    int mid = low + (high - low) / 2; // 避免溢出
    if (ST.elem[mid] == key) // 找到目标元素
    {
        return mid; // 返回位置（从0开始）
    }
    else if (ST.elem[mid] < key) // 目标在右半部分
    {
        return Search_Bin_Recursive(ST, key, mid + 1, high);
    }
    else // 目标在左半部分
    {
        return Search_Bin_Recursive(ST, key, low, mid - 1);
    }
}
```
