# 深度优先
```C
void DFS(ALGraph G,int v) 
{ /* 从第v个顶点出发递归地深度优先遍历图G。算法7.5 */ 
	visited[v] = 1;/* 设置访问标志为TRUE(已访问) */ 
	VisitFunc(G.vertices[v].data);/* 访问第v个顶点 */ 

    ArcNode *p = G.vertices[v].firstarc; // 获取第一个邻接点
    while (p != NULL) {
        int w = p->adjvex; // 邻接点的索引
        if (!visited[w]) { // 如果邻接点未被访问
            DFS(G, w); // 递归访问邻接点
        }
        p = p->nextarc; // 移动到下一个邻接点
    }
	/* 对v的尚未访问的邻接点w递归调用DFS */ 
} 
void DFSTraverse(ALGraph G) 
{ /* 对图G作深度优先遍历。算法7.4 */ 

	VisitFunc = print;/* 使用全局变量VisitFunc,使DFS不必设函数指针参数 */ 
    
    /* 访问标志数组初始化 */ 
    for (int i = 0; i < G.vexnum; i++) {
        visited[i] = 0;
    }
    
    /* 对尚未访问的顶点调用DFS */ 
    for (int i = 0; i < G.vexnum; i++) {
        if (!visited[i]) { // 如果顶点未被访问
            DFS(G, i); // 深度优先遍历
        }
    }
    
    printf("\n"); // 输出换行符
}
```
