```C++
#include <iostream>
#include <vector>
#include <algorithm> // 包含 next_permutation 函数

using namespace std;

int main() {
    int N;
    cin >> N; // 输入整数个数

    vector<int> nums(N);
    for (int i = 0; i < N; i++) {
        cin >> nums[i]; // 输入 N 个整数
    }

    // 对输入的整数进行排序，确保从最小的排列开始
    sort(nums.begin(), nums.end());

    // 使用 do-while 循环生成所有排列
    do {
        // 输出当前排列
        for (int i = 0; i < N; i++) {
            cout << nums[i] << (i == N - 1 ? "\n" : " ");
        }
    } while (next_permutation(nums.begin(), nums.end())); // 生成下一个排列

    return 0;
}
```
