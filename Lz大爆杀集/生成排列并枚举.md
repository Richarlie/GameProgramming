2.**用递归方法**
其思想就是：先输出所有以1为开头的排列（递归调用），然后输出以2为开头的排列（递归调用）······以此类推到以n为开头的排列
```C++
#include<stdio.h>
#include<stdlib.h>
#define MAXN 50
//O(N!)
void print_permutation(int n, int A[], int cur) { // cur == current
    if(cur == n) {
        for(int i=0; i<n; i++) printf("%d ",A[i]);
        putchar('\n');
    }
    else for(int i=1; i<=n; i++){
        int ok = 1;
        for (int j=0; j<cur; j++) if(A[j] == i) {ok = 0; break;}
        if(ok) {
            A[cur] = i;
            print_permutation(n, A, cur+1);
        }
    }
}

int main(){
    int n, A[MAXN];
    //scanf("%d",&n);
    n = 8;
    print_permutation(n, A, 0);
    return 0;
}
```
