# 二分插入函数
- 未有STL
```C++
#include <vector>

template<typename T>
void binary_insert_manual(std::vector<T>& vec, const T& item) {
    if (vec.empty()) {
        vec.push_back(item);
        return;
    }
    
    size_t left = 0;
    size_t right = vec.size();
    
    // 二分查找确定插入位置
    while (left < right) {
        size_t mid = left + (right - left) / 2;
        if (vec[mid] < item) {
            left = mid + 1;
        } else {
            right = mid;
        }
    }
    
    // 在找到的位置插入元素
    vec.insert(vec.begin() + left, item);
}
```
